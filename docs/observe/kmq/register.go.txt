package prom

import (
	"sync"

	"github.com/prometheus/client_golang/prometheus"
)

var regOnce sync.Once

func MustRegisterAll() {
	regOnce.Do(func() {
		// kmq
		prometheus.MustRegister(KmqGateBacklog)
		prometheus.MustRegister(KmqBatchCommitTotal)
		prometheus.MustRegister(KmqBatchCommitFailedTotal)
		prometheus.MustRegister(KmqRebalanceTotal)
		prometheus.MustRegister(KmqDLQMessagesTotal)
	})
}
